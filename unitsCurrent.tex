\begin{equation}\label{eqn:unitsCurrent:20}
[M] = \frac{V}{m^2}
\end{equation}
\begin{equation}\label{eqn:unitsCurrent:40}
\begin{aligned}
[ \eta J ]
&= \Omega \frac{A}{m^2} \\
&= \frac{V}{A} \frac{A}{m^2} \\
&= \frac{V}{m^2}
\end{aligned}
\end{equation}
\begin{equation}\label{eqn:unitsCurrent:60}
\begin{aligned}
[ \eta c \rho ]
&= \Omega \frac{m}{s} \frac{C}{m^2}  \\
&= \Omega \frac{A}{m^2}  \\
&= \frac{V}{m^2}
\end{aligned}
\end{equation}
\begin{equation}\label{eqn:unitsCurrent:80}
\begin{aligned}
[ c \rho_m ]
&= \frac{m}{s} \frac{W}{m^3} \\
&= \frac{W}{s m^2} \\
&= \frac{[B L^2]}{s m^2} \\
&= \frac{[\mu H]}{s} \\
&= \inv{s} \frac{N}{A^2} \frac{A}{m} \\
&= \inv{s m} \frac{N}{A} \\
&= \inv{s m} \frac{C V}{A m} \\
&= \inv{m} \frac{V}{m} \\
&= \frac{V}{m^2} \\
\end{aligned}
\end{equation}
