\begin{equation*}
\begin{aligned}
A \rfloor B &= \sum_{i,j = 0}^N \gpgrade{ A_i B_j }{j - i} \\
A \lfloor B &= \sum_{i,j = 0}^N \gpgrade{ A_i B_j }{i - j} \\
A \cdot B &= \sum_{i,j = 0}^N \gpgrade{ A_i B_j }{\Abs{i - j}}.
\end{aligned}
\end{equation*}

\begin{equation}\label{eqn:contraction_vs_dot:n}
\Proj_{\Ba}(\Bb) = \lr{\Bb \wedge \Ba} \cdot \inv{\Ba} = \lr{\Bb \wedge \Ba} \lfloor \inv{\Ba} = \inv{\Ba} \rfloor \lr{\Ba \wedge \Bb} 
\end{equation}
